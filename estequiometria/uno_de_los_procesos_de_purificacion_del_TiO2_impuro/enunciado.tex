    \ce{TiO2(impuro) + C(s) + Cl2(g) -> TiCl4(g) + CO2(g) + CO(g)}\\
    \ce{TiCl4(g) + O2(g) -> TiO2(s) + Cl2(g)}
\resultadocmd{ \SI{226}{\gram} }

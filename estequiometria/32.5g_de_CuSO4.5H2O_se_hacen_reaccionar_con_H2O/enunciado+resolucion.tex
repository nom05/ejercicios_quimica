\begin{frame}
	\frametitle{\ejerciciocmd}
	\framesubtitle{Resolución (\rom{1}): determinación de la concentración molar de \ce{CuSO4*5H2O}}
	\structure{Número de moles de cada sustancia:} $n=\rfrac{m}{Mm}\Rightarrow m=n\vdot Mm$
	$$
		n(\ce{CuSO4*\colorbox{red}{\color{white}\textbf{\num{5}}}H2O}) = \frac{\SI{32,5}{\cancel\gram}}{\SI{249,6854}{\cancel\gram\per\mol}} = \SI{,1302}{\mol} = n(\ce{CuSO4}) \Rightarrow
	$$
	Usando la ecuación de los gases ideales: $PV=nRT$
	$$
		n=\frac{P\vdot V}{R\vdot T}\Rightarrow n(\ce{H2O})_{\text{vapor}}=\frac{\rfrac{732}{760}~\si{\cancel\atm}\vdot\SI{1200}{\cancel\liter}}{\SI{,082}{\cancel\atm\cancel\liter\per\mol\per\cancel\kelvin}\vdot\SI{358,15}{\cancel\kelvin}} = \SI{39,355}{\mol}
	$$
	$$
		m(\ce{H2O}) = \SI{39,355}{\cancel\mol}\vdot\SI{18,02}{\gram\per\cancel\mol} = \SI{709,18}{\gram} = \SI{,709}{\kilogram}
	$$
